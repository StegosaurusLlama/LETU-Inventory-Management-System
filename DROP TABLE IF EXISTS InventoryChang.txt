DROP TABLE IF EXISTS InventoryChange;
DROP TABLE IF EXISTS Users;
DROP TABLE IF EXISTS ProductTag;
DROP TABLE IF EXISTS Tag;
DROP TABLE IF EXISTS StockItem;

CREATE TABLE Users (
  userID INTEGER PRIMARY KEY AUTOINCREMENT, 
  username TEXT NOT NULL UNIQUE, 
  passwordHash TEXT NOT NULL,
  clearance INTEGER NOT NULL
);

CREATE TABLE InventoryChange (
  changeID INTEGER PRIMARY KEY AUTOINCREMENT, 
  changeTime TEXT NOT NULL DEFAULT (datetime('now')), 
  userID INTEGER NOT NULL,
  actionTaken TEXT NOT NULL,
  effectiveChange TEXT NOT NULL,
  changeQuantity INTEGER NOT NULL,
  FOREIGN KEY (userID) REFERENCES Users(userID)
);

CREATE TABLE Tag (
  tagID INTEGER PRIMARY KEY AUTOINCREMENT, 
  name TEXT NOT NULL UNIQUE
);

CREATE TABLE StockItem (
  productID INTEGER PRIMARY KEY AUTOINCREMENT, 
  name TEXT NOT NULL UNIQUE, 
  price REAL NOT NULL,
  quantity INTEGER NOT NULL,
  description TEXT NOT NULL,
  lowThreshhold INTEGER NOT NULL,
  image TEXT
);

CREATE TABLE ProductTag (
  productID INTEGER NOT NULL,
  tagID INTEGER NOT NULL,
  FOREIGN KEY (productID) REFERENCES StockItem(productID) ON DELETE CASCADE,
  FOREIGN KEY (tagID) REFERENCES Tag(tagID) ON DELETE CASCADE,
  PRIMARY KEY (productID, tagID)
);

CREATE TRIGGER no_log_delete
BEFORE DELETE ON InventoryChange
BEGIN
    SELECT RAISE(ABORT, 'Deletes not allowed on InventoryChange');
END;
